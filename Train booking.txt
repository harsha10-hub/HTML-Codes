        { id: 7, name: "Kolkata Mail", from: "Mumbai", to: "Kolkata", departure: "22:00", arrival: "05:00", duration: "31h", price: 1250, class: "Sleeper", seatsTotal: 110, bookedSeats: ["S45","S46"], img: "https://images.unsplash.com/photo-1513094735237-8f2714d57c13?w=500&auto=format" }
    ];

    let currentFrom = "Delhi";
    let currentTo = "Mumbai";
    let currentClass = "all";
    let filteredTrains = [];
    let currentSelectedTrain = null;
    let selectedSeatsArray = [];
    let userBookings = JSON.parse(localStorage.getItem("railyatri_bookings")) || [];
    
    // Coach & seat generation
    let currentCoach = "Sleeper"; // default
    const coachSeatMap = {
        "Sleeper": { prefix: "S", rows: 10, cols: 6, priceMultiplier: 1 },
        "3A": { prefix: "A", rows: 8, cols: 6, priceMultiplier: 1.2 },
        "2A": { prefix: "B", rows: 6, cols: 5, priceMultiplier: 1.5 },
        "CC": { prefix: "C", rows: 9, cols: 5, priceMultiplier: 0.9 }
    };

    function generateSeatLabels(trainClass, trainObj) {
        const config = coachSeatMap[trainClass];
        if (!config) return [];
        let seats = [];
        for(let i=1; i<=config.rows; i++) {
            for(let j=1; j<=config.cols; j++) {
                seats.push(`${config.prefix}${i}${j}`);
            }
        }
        return seats.slice(0, trainObj.seatsTotal);
    }

    function renderTrains() {
        let filtered = trainsData.filter(train => {
            const matchRoute = train.from === currentFrom && train.to === currentTo;
            const matchClass = currentClass === "all" || train.class === currentClass;
            return matchRoute && matchClass;
        });
        filteredTrains = filtered;
        const sortBy = document.getElementById("sortSelect").value;
        if (sortBy === "price") filteredTrains.sort((a,b) => a.price - b.price);
        if (sortBy === "departure") filteredTrains.sort((a,b) => a.departure.localeCompare(b.departure));
        if (sortBy === "duration") filteredTrains.sort((a,b) => a.duration.localeCompare(b.duration));
        
        const grid = document.getElementById("trainGrid");
        document.getElementById("resultCountSpan").innerText = filteredTrains.length;
        if (filteredTrains.length === 0) {
            grid.innerHTML = `<div style="grid-column:1/-1; text-align:center; padding:50px;">🚂 No trains found from ${currentFrom} to ${currentTo}. Try different route.</div>`;
            return;
        }
        grid.innerHTML = filteredTrains.map(train => {
            const availableSeats = train.seatsTotal - train.bookedSeats.length;
            return `
            <div class="train-card" data-id="${train.id}">
                <div class="train-img" style="background-image: url('${train.img}');">
                    <div class="train-type">${train.class}</div>
                </div>
                <div class="train-info">
                    <div class="train-name">${train.name} <span class="price">₹${train.price}</span></div>
                    <div class="route-details">
                        <span><i class="far fa-clock"></i> ${train.departure}</span>
                        <span>→</span>
                        <span>${train.arrival}</span>
                    </div>
                    <div style="display: flex; justify-content: space-between; align-items:center;">
                        <span>⏱️ ${train.duration}</span>
                        <span class="seats-left"><i class="fas fa-chair"></i> ${availableSeats} seats left</span>
                    </div>
                    <button class="book-btn" data-id="${train.id}">Book Tickets <i class="fas fa-chevron-right"></i></button>
                </div>
            </div>`;
        }).join('');

        document.querySelectorAll('.book-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.dataset.id);
                const train = filteredTrains.find(t => t.id === id);
                if (train) openSeatModal(train);
            });
        });
    }

    function openSeatModal(train) {
        currentSelectedTrain = train;
        selectedSeatsArray = [];
        currentCoach = train.class;
        document.getElementById("modalTrainTitle").innerHTML = `${train.name} (${train.from} → ${train.to}) &nbsp; <span style="font-size:0.8rem;">${train.class}</span>`;
        renderCoachSelector(train);
        renderSeatMap(train);
        updateModalSummary();
        document.getElementById("seatModal").classList.add("active");
    }

    function renderCoachSelector(train) {
        const coachDiv = document.getElementById("coachSelector");
        const classes = ["Sleeper", "3A", "2A", "CC"];
        coachDiv.innerHTML = classes.map(cls => `
            <button class="coach-btn ${currentCoach === cls ? 'active' : ''}" data-coach="${cls}">${cls}</button>
        `).join('');
        document.querySelectorAll('.coach-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                currentCoach = btn.dataset.coach;
                // Update price based on coach multiplier
                renderCoachSelector(train);
                renderSeatMap(train);
                updateModalSummary();
            });
        });
    }

    function renderSeatMap(train) {
        const allSeats = generateSeatLabels(currentCoach, train);
        const booked = train.bookedSeats || [];
        const seatContainer = document.getElementById("seatMapGrid");
        seatContainer.innerHTML = "";
        allSeats.forEach(seat => {
            const isBooked = booked.includes(seat);
            const isSelected = selectedSeatsArray.includes(seat);
            const seatDiv = document.createElement("div");
            seatDiv.classList.add("seat");
            if (isBooked) seatDiv.classList.add("booked");
            if (isSelected) seatDiv.classList.add("selected");
            seatDiv.innerText = seat;
            if (!isBooked) {
                seatDiv.classList.add("available");
                seatDiv.addEventListener("click", () => {
                    if (selectedSeatsArray.includes(seat)) {
                        selectedSeatsArray = selectedSeatsArray.filter(s => s !== seat);
                    } else {
                        selectedSeatsArray.push(seat);
                    }
                    renderSeatMap(train);
                    updateModalSummary();
                });
            }
            seatContainer.appendChild(seatDiv);
        });
    }

    function updateModalSummary() {
        const count = selectedSeatsArray.length;
        const multiplier = coachSeatMap[currentCoach]?.priceMultiplier || 1;
        const basePrice = currentSelectedTrain ? currentSelectedTrain.price : 0;
        const total = Math.round(count * basePrice * multiplier);
        document.getElementById("selectedSeatsLabel").innerText = count ? selectedSeatsArray.join(", ") : "None";
        document.getElementById("totalFareSpan").innerText = total;
    }

    function confirmBooking() {
        if (!currentSelectedTrain) return;
        if (selectedSeatsArray.length === 0) {
            alert("Please select at least one seat.");
            return;
        }
        const conflict = selectedSeatsArray.some(seat => currentSelectedTrain.bookedSeats.includes(seat));
        if (conflict) {
            alert("Some seats are already booked. Please reselect.");
            openSeatModal(currentSelectedTrain);
            return;
        }
        const multiplier = coachSeatMap[currentCoach]?.priceMultiplier || 1;
        const totalPrice = Math.round(selectedSeatsArray.length * currentSelectedTrain.price * multiplier);
        const bookingRecord = {
            id: Date.now(),
            trainId: currentSelectedTrain.id,
            trainName: currentSelectedTrain.name,
            from: currentSelectedTrain.from,
            to: currentSelectedTrain.to,
            class: currentCoach,
            date: document.getElementById("travelDate").value,
            seats: [...selectedSeatsArray],
            totalPrice: totalPrice,
            bookedOn: new Date().toLocaleString()
        };
        userBookings.push(bookingRecord);
        localStorage.setItem("railyatri_bookings", JSON.stringify(userBookings));
        
        currentSelectedTrain.bookedSeats.push(...selectedSeatsArray);
        const index = trainsData.findIndex(t => t.id === currentSelectedTrain.id);
        if (index !== -1) trainsData[index].bookedSeats = currentSelectedTrain.bookedSeats;
        localStorage.setItem("railyatri_trains_state", JSON.stringify(trainsData.map(t => ({ id: t.id, bookedSeats: t.bookedSeats }))));
        
        alert(`✅ Booking Confirmed!\nTrain: ${currentSelectedTrain.name}\nClass: ${currentCoach}\nSeats: ${selectedSeatsArray.join(", ")}\nTotal: ₹${totalPrice}\nJourney: ${currentSelectedTrain.from} → ${currentSelectedTrain.to} on ${bookingRecord.date}`);
        
        closeModalAndRefresh();
    }

    function closeModalAndRefresh() {
        document.getElementById("seatModal").classList.remove("active");
        currentSelectedTrain = null;
        selectedSeatsArray = [];
        renderTrains();
    }

    function showMyTrips() {
        if (userBookings.length === 0) {
            alert("🚂 No bookings yet. Book your first train ticket!");
            return;
        }
        let msg = "🎫 YOUR BOOKED TRAIN TICKETS:\n\n";
        userBookings.forEach((b, idx) => {
            msg += `${idx+1}. ${b.trainName} | ${b.from} → ${b.to}\n   Class: ${b.class} | Seats: ${b.seats.join(", ")} | ₹${b.totalPrice} | Date: ${b.date}\n`;
        });
        alert(msg);
    }

    function loadPersistedSeats() {
        const savedState = localStorage.getItem("railyatri_trains_state");
        if (savedState) {
            const state = JSON.parse(savedState);
            state.forEach(updated => {
                const train = trainsData.find(t => t.id === updated.id);
                if (train) train.bookedSeats = updated.bookedSeats;
            });
        }
        const storedBookings = localStorage.getItem("railyatri_bookings");
        if (storedBookings) userBookings = JSON.parse(storedBookings);
    }
