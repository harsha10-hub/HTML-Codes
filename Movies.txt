<script>
    // MOVIE DATABASE with images (from unsplash)
    const movies = [
        { id: 1, title: "Interstellar", genre: "Sci-Fi", rating: 8.7, year: "2014", duration: "2h 49m", img: "https://images.unsplash.com/photo-1534809027769-b00d750a6bac?w=500&auto=format", bookedSeats: ["A3","B5"] },
        { id: 2, title: "John Wick 4", genre: "Action", rating: 8.2, year: "2023", duration: "2h 49m", img: "https://images.unsplash.com/photo-1626814026160-2237a95fc5a0?w=500&auto=format", bookedSeats: ["C1","C2"] },
        { id: 3, title: "The Batman", genre: "Action", rating: 8.0, year: "2022", duration: "2h 56m", img: "https://images.unsplash.com/photo-1509347528160-9a9e33742cdb?w=500&auto=format", bookedSeats: ["D4"] },
        { id: 4, title: "Dune: Part Two", genre: "Adventure", rating: 8.9, year: "2024", duration: "2h 46m", img: "https://images.unsplash.com/photo-1587923623987-c7e4083beb23?w=500&auto=format", bookedSeats: ["B2","B3"] },
        { id: 5, title: "Oppenheimer", genre: "Drama", rating: 8.6, year: "2023", duration: "3h", img: "https://images.unsplash.com/photo-1440404653325-ab127d49abc1?w=500&auto=format", bookedSeats: ["E1","E5"] },
        { id: 6, title: "Inception", genre: "Sci-Fi", rating: 8.8, year: "2010", duration: "2h 28m", img: "https://images.unsplash.com/photo-1485846234645-a62644f84728?w=500&auto=format", bookedSeats: ["A1"] }
    ];

    // Predefined seat layout (rows A-E, 1-6)
    const seatLabels = ["A1","A2","A3","A4","A5","A6", "B1","B2","B3","B4","B5","B6", "C1","C2","C3","C4","C5","C6", "D1","D2","D3","D4","D5","D6", "E1","E2","E3","E4","E5","E6"];

    let currentMovie = null;
    let selectedSeats = [];
    let userBookings = JSON.parse(localStorage.getItem("cinebookings")) || []; // store {movieId, movieTitle, seats, total}

    // render movie grid based on filter/search
    let currentGenre = "all";
    let currentSearch = "";

    function renderMovies() {
        let filtered = movies.filter(m => {
            const matchGenre = currentGenre === "all" || m.genre === currentGenre;
            const matchSearch = m.title.toLowerCase().includes(currentSearch.toLowerCase());
            return matchGenre && matchSearch;
        });
        const grid = document.getElementById("movieGrid");
        if (filtered.length === 0) {
            grid.innerHTML = `<div style="grid-column:1/-1; text-align:center; padding:50px;">🎞️ No movies found. Try another genre.</div>`;
            return;
        }
        grid.innerHTML = filtered.map(movie => `
            <div class="movie-card" data-id="${movie.id}">
                <div class="movie-poster" style="background-image: url('${movie.img}');">
                    <div class="rating-badge"><i class="fas fa-star"></i> ${movie.rating} / 10</div>
                </div>
                <div class="movie-info">
                    <div class="movie-title">${movie.title} <span style="font-size:0.8rem;">${movie.year}</span></div>
                    <div class="movie-genre">${movie.genre} • ${movie.duration}</div>
                    <button class="book-btn" data-id="${movie.id}"><i class="fas fa-ticket-alt"></i> Book Tickets</button>
                </div>
            </div>
        `).join('');
        // attach booking events
        document.querySelectorAll('.book-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.dataset.id);
                const movie = movies.find(m => m.id === id);
                if (movie) openBookingModal(movie);
                e.stopPropagation();
            });
        });
    }

    // open seat selection modal
    function openBookingModal(movie) {
        currentMovie = movie;
        selectedSeats = [];
        document.getElementById("modalMovieTitle").innerText = movie.title;
        generateSeatGrid(movie.bookedSeats);
        updateModalSummary();
        document.getElementById("bookingModal").classList.add("active");
    }

    function generateSeatGrid(bookedSeatsArray) {
        const seatContainer = document.getElementById("seatGrid");
        seatContainer.innerHTML = "";
        seatLabels.forEach(seat => {
            const isBooked = bookedSeatsArray.includes(seat);
            const isSelected = selectedSeats.includes(seat);
            const seatDiv = document.createElement("div");
            seatDiv.classList.add("seat");
            if (isBooked) seatDiv.classList.add("booked");
            if (isSelected) seatDiv.classList.add("selected");
            seatDiv.innerText = seat;
            if (!isBooked) {
                seatDiv.classList.add("available");
                seatDiv.addEventListener("click", () => {
                    if (selectedSeats.includes(seat)) {
                        selectedSeats = selectedSeats.filter(s => s !== seat);
                    } else {
                        selectedSeats.push(seat);
                    }
                    generateSeatGrid(currentMovie.bookedSeats);
                    updateModalSummary();
                });
            }
            seatContainer.appendChild(seatDiv);
        });
    }

    function updateModalSummary() {
        const count = selectedSeats.length;
        const total = count * 180;
        document.getElementById("selectedSeatsDisplay").innerText = count ? selectedSeats.join(", ") : "None";
        document.getElementById("totalPriceSpan").innerHTML = `₹${total}`;
    }

    function confirmBooking() {
        if (!currentMovie) return;
        if (selectedSeats.length === 0) {
            alert("Please select at least one seat.");
            return;
        }
        // check if seats are still available (avoid race, but static)
        const conflict = selectedSeats.some(seat => currentMovie.bookedSeats.includes(seat));
        if (conflict) {
            alert("Some of your selected seats are no longer available. Please reselect.");
            openBookingModal(currentMovie);
            return;
        }
        // add to local storage booking
        const bookingRecord = {
            id: Date.now(),
            movieId: currentMovie.id,
            movieTitle: currentMovie.title,
            seats: [...selectedSeats],
            total: selectedSeats.length * 180,
            date: new Date().toLocaleString()
        };
        userBookings.push(bookingRecord);
        localStorage.setItem("cinebookings", JSON.stringify(userBookings));
        // update movie's booked seats (persist in memory + local)
        currentMovie.bookedSeats.push(...selectedSeats);
        // update static movies array in memory, but also store updated booked seats to localStorage for "my bookings" reflection
        const movieIndex = movies.findIndex(m => m.id === currentMovie.id);
        if(movieIndex !== -1) movies[movieIndex].bookedSeats = currentMovie.bookedSeats;
        // store movies state into localStorage for persistence
        localStorage.setItem("cine_movies_state", JSON.stringify(movies.map(m => ({ id: m.id, bookedSeats: m.bookedSeats }))));
        
        alert(`🎉 Booking confirmed!\nMovie: ${currentMovie.title}\nSeats: ${selectedSeats.join(", ")}\nTotal: ₹${selectedSeats.length * 180}\nScreenshot this for check-in.`);
        closeModal();
        renderMovies(); // refresh grid (booked buttons state visual)
    }

    function closeModal() {
        document.getElementById("bookingModal").classList.remove("active");
        currentMovie = null;
        selectedSeats = [];
    }

    // My Bookings panel (simple modal or alert)
    function showMyBookings() {
        if (userBookings.length === 0) {
            alert("📭 You have no bookings yet. Book your first movie ticket!");
            return;
        }
        let message = "🎟️ YOUR BOOKINGS:\n\n";
        userBookings.forEach((b, idx) => {
            message += `${idx+1}. ${b.movieTitle} | Seats: ${b.seats.join(", ")} | ₹${b.total} | ${b.date}\n`;
        });
        alert(message);
    }

    // load persisted booked seats from localStorage (sync)
    function loadPersistedBookings() {
        const savedState = localStorage.getItem("cine_movies_state");
        if (savedState) {
            const state = JSON.parse(savedState);
            state.forEach(updated => {
                const movie = movies.find(m => m.id === updated.id);
                if (movie) movie.bookedSeats = updated.bookedSeats;
            });
        }
        // also load user bookings array
        const storedBookings = localStorage.getItem("cinebookings");
        if (storedBookings) userBookings = JSON.parse(storedBookings);
    }

    // event listeners
    document.getElementById("movieSearch").addEventListener("input", (e) => {
        currentSearch = e.target.value;
        renderMovies();
    });
    document.getElementById("genreFilter").addEventListener("change", (e) => {
        currentGenre = e.target.value;
        renderMovies();
    });
    document.getElementById("resetFilterBtn").addEventListener("click", () => {
        document.getElementById("movieSearch").value = "";
        document.getElementById("genreFilter").value = "all";
        currentSearch = "";
        currentGenre = "all";
        renderMovies();
    });
    document.getElementById("exploreBtn").addEventListener("click", () => {
        document.querySelector(".movie-grid").scrollIntoView({ behavior: "smooth" });
    });
    document.getElementById("myBookingsBtn").addEventListener("click", showMyBookings);
    document.getElementById("confirmBookingBtn").addEventListener("click", confirmBooking);
    document.getElementById("closeModalBtn").addEventListener("click", closeModal);
    document.getElementById("bookingModal").addEventListener("click", (e) => {
        if (e.target === document.getElementById("bookingModal")) closeModal();
    });

    // initial load
    loadPersistedBookings();
    renderMovies();
</script>
