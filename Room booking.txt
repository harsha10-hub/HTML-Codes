    function renderRooms() {
        const container = document.getElementById('roomsContainer');
        if (!container) return;
        container.innerHTML = '';
        roomsData.forEach(room => {
            const card = document.createElement('div');
            card.className = 'room-card';
            card.innerHTML = `
                <img class="room-img" src="${room.img}" alt="${room.name}" loading="lazy">
                <div class="room-info">
                    <h3>${room.name}</h3>
                    <div class="room-details"><span><i class="fas fa-expand"></i> ${room.size}</span><span><i class="fas fa-bed"></i> premium comfort</span></div>
                    <p style="color: #6e5742; margin: 0.5rem 0;">${room.desc}</p>
                    <div class="price">$${room.price} <small>/ night</small></div>
                    <button class="book-btn" data-roomname="${room.name}"><i class="fas fa-calendar-plus"></i> Quick book</button>
                </div>
            `;
            container.appendChild(card);
        });

        // attach quick book listeners
        document.querySelectorAll('.book-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const roomName = btn.getAttribute('data-roomname');
                const selectElem = document.getElementById('roomTypeSelect');
                if (selectElem) {
                    // try to match room
                    let found = false;
                    for(let i=0; i<selectElem.options.length; i++) {
                        if(selectElem.options[i].value === roomName) {
                            selectElem.selectedIndex = i;
                            found = true;
                            break;
                        }
                    }
                    if(!found) {
                        // fallback
                        selectElem.value = roomName;
                    }
                }
                // scroll to booking form
                document.querySelector('.booking-panel').scrollIntoView({ behavior: 'smooth', block: 'center' });
                // show small notification
                const msgDiv = document.getElementById('bookingMsg');
                if(msgDiv) {
                    msgDiv.style.display = 'block';
                    msgDiv.innerHTML = `✨ "${roomName}" selected. Fill details to confirm.`;
                    msgDiv.style.background = "#f0e2d4";
                    setTimeout(() => {
                        if(msgDiv) msgDiv.style.opacity = '0';
                        setTimeout(() => {
                            if(msgDiv) msgDiv.style.display = 'none';
                            if(msgDiv) msgDiv.style.opacity = '1';
                        }, 1500);
                    }, 2000);
                }
            });
        });
    }

    // Booking form handling
    const form = document.getElementById('roomBookingForm');
    const msgBox = document.getElementById('bookingMsg');

    function showBookingMessage(text, isError = false) {
        msgBox.style.display = 'block';
        msgBox.innerHTML = text;
        msgBox.style.backgroundColor = isError ? '#ffefed' : '#e2f0e6';
        msgBox.style.color = isError ? '#b95c3a' : '#2f6b47';
        setTimeout(() => {
            msgBox.style.opacity = '0';
            setTimeout(() => {
                if(msgBox.style.display !== 'none') msgBox.style.display = 'none';
                msgBox.style.opacity = '1';
            }, 500);
        }, 3500);
    }

    form.addEventListener('submit', function(e) {
        e.preventDefault();

        const name = document.getElementById('guestName').value.trim();
        const email = document.getElementById('guestEmail').value.trim();
        const room = document.getElementById('roomTypeSelect').value;
        const checkin = document.getElementById('checkin').value;
        const checkout = document.getElementById('checkout').value;
        const guests = document.getElementById('guests').value;

        if(!name) { showBookingMessage("Please enter your full name.", true); return; }
        if(!email || !email.includes('@')) { showBookingMessage("Valid email required.", true); return; }
        if(!checkin || !checkout) { showBookingMessage("Select check-in and check-out dates.", true); return; }
        const checkinDate = new Date(checkin);
        const checkoutDate = new Date(checkout);
        if(checkoutDate <= checkinDate) { showBookingMessage("Check-out must be after check-in.", true); return; }

        // simulate booking reference
        const bookingRef = 'HR-' + Math.floor(Math.random() * 90000 + 10000);
        const nights = Math.ceil((checkoutDate - checkinDate) / (1000 * 60 * 60 * 24));
        showBookingMessage(`✅ Booking confirmed, ${name}! ${room} reserved for ${nights} night(s). Reference: ${bookingRef}. A confirmation sent to ${email}. Pay at property.`, false);
        // optional reset
