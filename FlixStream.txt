     Render rows
    function renderRows() {
        const trendingMovies = movieDatabase.slice(0, 5);
        const popularMovies = movieDatabase.slice(2, 8);
        const actionMovies = movieDatabase.filter(m = m.genre === Action).slice(0, 6);
        
        document.getElementById(trendingRow).innerHTML = trendingMovies.map(m = createMovieCard(m)).join();
        document.getElementById(popularRow).innerHTML = popularMovies.map(m = createMovieCard(m)).join();
        document.getElementById(actionRow).innerHTML = actionMovies.map(m = createMovieCard(m)).join();
        
         Attach click events on all movie cards
        document.querySelectorAll(.movie-card).forEach(card = {
            card.addEventListener(click, (e) = {
                const id = parseInt(card.dataset.id);
                const movie = movieDatabase.find(m = m.id === id);
                if (movie) openModalWithMovie(movie);
                e.stopPropagation();
            });
        });
    }
