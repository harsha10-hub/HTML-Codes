    // Cart state
    let cart = [];  // each item { id, name, price, quantity }

    // Helper: update UI (cart badge, total, cart items list)
    function updateCartUI() {
        const cartCount = cart.reduce((acc, item) => acc + item.quantity, 0);
        document.getElementById('cartCountBadge').innerText = cartCount;
        const cartTotal = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
        document.getElementById('cartTotalPrice').innerText = `₹${cartTotal}`;

        const cartContainer = document.getElementById('cartItemsList');
        if (cart.length === 0) {
            cartContainer.innerHTML = `<div style="text-align:center; color:#aaa; padding:30px;"><i class="fas fa-shopping-basket"></i> Your cart is empty</div>`;
        } else {
            cartContainer.innerHTML = cart.map(item => `
                <div class="cart-item" data-id="${item.id}">
                    <div class="cart-item-info">
                        <h4>${item.name}</h4>
                        <div class="cart-item-price">₹${item.price}</div>
                    </div>
                    <div class="cart-item-actions">
                        <button class="decrCartBtn" data-id="${item.id}">-</button>
                        <span style="margin:0 8px; font-weight:600;">${item.quantity}</span>
                        <button class="incrCartBtn" data-id="${item.id}">+</button>
                        <button class="removeCartBtn" data-id="${item.id}" style="background:#ffefe5; border:none; margin-left:6px;"><i class="fas fa-trash-alt"></i></button>
                    </div>
                </div>
            `).join('');
        }
        attachCartItemEvents();
    }

    function attachCartItemEvents() {
        document.querySelectorAll('.decrCartBtn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.dataset.id);
                const idx = cart.findIndex(i => i.id === id);
                if (idx !== -1) {
                    if (cart[idx].quantity > 1) cart[idx].quantity--;
                    else cart.splice(idx, 1);
                    updateCartUI();
                    saveCartToLocal();
                }
            });
        });
        document.querySelectorAll('.incrCartBtn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.dataset.id);
                const idx = cart.findIndex(i => i.id === id);
                if (idx !== -1) cart[idx].quantity++;
                updateCartUI();
                saveCartToLocal();
            });
        });
        document.querySelectorAll('.removeCartBtn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.dataset.id);
                cart = cart.filter(i => i.id !== id);
                updateCartUI();
                saveCartToLocal();
            });
        });
    }

    function addToCart(menuItem) {
        const existing = cart.find(i => i.id === menuItem.id);
        if (existing) {
            existing.quantity++;
        } else {
            cart.push({ id: menuItem.id, name: menuItem.name, price: menuItem.price, quantity: 1 });
        }
        updateCartUI();
        saveCartToLocal();
        // subtle feedback
        const btn = document.querySelector(`.order-btn[data-id='${menuItem.id}']`);
        if(btn) {
            btn.style.transform = "scale(0.97)";
            setTimeout(()=> btn.style.transform = "", 150);
        }
    }

    function saveCartToLocal() {
        localStorage.setItem('flavor_cart', JSON.stringify(cart));
    }

    function loadCartFromLocal() {
        const saved = localStorage.getItem('flavor_cart');
        if (saved) {
            try {
                cart = JSON.parse(saved);
                updateCartUI();
            } catch(e) { cart = []; }
        }
    }

    // Render menu cards based on filters (search & category)
    let currentFilterCategory = "all";
    let currentSearchTerm = "";

    function renderMenu() {
        const filtered = menuData.filter(item => {
            const matchesCategory = (currentFilterCategory === "all" || item.category === currentFilterCategory);
            const matchesSearch = item.name.toLowerCase().includes(currentSearchTerm.toLowerCase());
            return matchesCategory && matchesSearch;
        });
        const grid = document.getElementById('menuGrid');
        if (filtered.length === 0) {
            grid.innerHTML = `<div style="grid-column:1/-1; text-align:center; padding:60px;"><i class="fas fa-utensils fa-2x" style="color:#e67e22;"></i><p style="margin-top:12px;">No dishes match. Try something else!</p></div>`;
            return;
        }
        grid.innerHTML = filtered.map(item => `
            <div class="food-card">
                <div class="card-img" style="background-image: url('${item.img}'); background-size: cover;">
                    <div class="badge">${item.category}</div>
                </div>
                <div class="card-content">
                    <div class="card-header">
                        <h3>${item.name}</h3>
                        <span class="price">₹${item.price}</span>
                    </div>
                    <p class="desc">${item.desc}</p>
                    <button class="order-btn" data-id="${item.id}"><i class="fas fa-plus-circle"></i> Add to Cart</button>
                </div>
            </div>
        `).join('');

        // attach add to cart events
        document.querySelectorAll('.order-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.getAttribute('data-id'));
                const foundItem = menuData.find(m => m.id === id);
                if (foundItem) addToCart(foundItem);
            });
        });
    }

    // filter handlers
    document.getElementById('searchInput').addEventListener('input', (e) => {
        currentSearchTerm = e.target.value;
        renderMenu();
    });
    document.getElementById('categoryFilter').addEventListener('change', (e) => {
        currentFilterCategory = e.target.value;
        renderMenu();
    });
    document.getElementById('resetFiltersBtn').addEventListener('click', () => {
        document.getElementById('searchInput').value = "";
        document.getElementById('categoryFilter').value = "all";
        currentSearchTerm = "";
        currentFilterCategory = "all";
        renderMenu();
    });

    // cart panel handlers

        cartPanel.classList.remove('open');
        overlay.classList.remove('active');
    }

            alert("Your cart is empty! Add some delicious items first.");
            return;
        }
        alert(`✅ Order placed! Total ₹${cart.reduce((s,i)=>s+(i.price*i.quantity),0)}. Thank you for ordering from FlavorHaven.`);

        updateCartUI();
        saveCartToLocal();
        closeCart();
    });
