<script>
    // dynamic destinations for india (flight suggestive)
    const destinations = [
        { name: "Goa", img: "https://images.pexels.com/photos/2728912/beach-sunset-sea-ocean-2728912.jpg?auto=compress&cs=tinysrgb&w=400", price: "from ₹2,499", desc: "Sun, sand & parties" },
        { name: "Jaipur", img: "https://images.pexels.com/photos/2044292/amber-fort-jaipur-2044292.jpg?auto=compress&cs=tinysrgb&w=400", price: "from ₹3,199", desc: "Pink City palaces" },
        { name: "Varanasi", img: "https://images.pexels.com/photos/3176082/varanasi-ganga-aarti-3176082.jpg?auto=compress&cs=tinysrgb&w=400", price: "from ₹3,899", desc: "Ganga ghats" },
        { name: "Kochi", img: "https://images.pexels.com/photos/5676441/backwaters-kerala-5676441.jpg?auto=compress&cs=tinysrgb&w=400", price: "from ₹3,549", desc: "God's own country" }
    ];

    function renderDestinations() {
        const grid = document.getElementById('destinationsGrid');
        if (!grid) return;
        grid.innerHTML = '';
        destinations.forEach(dest => {
            const card = document.createElement('div');
            card.className = 'dest-card';
            card.innerHTML = `
                <img class="dest-img" src="${dest.img}" alt="${dest.name}" loading="lazy">
                <div class="dest-info">
                    <h3>${dest.name}</h3>
                    <p style="color:#5d7184;">${dest.desc}</p>
                    <div class="price-tag">${dest.price}</div>
                    <button class="select-flight" data-dest="${dest.name}"><i class="fas fa-ticket-alt"></i> View flights</button>
                </div>
            `;
            grid.appendChild(card);
        });
        // attach events after
        document.querySelectorAll('.select-flight').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const destName = btn.getAttribute('data-dest');
                const toSelect = document.getElementById('toCity');
                // fuzzy match to select city
                const options = toSelect.options;
                for (let i = 0; i < options.length; i++) {
                    if (options[i].value.toLowerCase().includes(destName.toLowerCase())) {
                        toSelect.selectedIndex = i;
                        break;
                    } else if (destName === "Goa" && options[i].value.includes("GOI")) toSelect.selectedIndex = i;
                    else if (destName === "Jaipur" && options[i].value.includes("JAI")) toSelect.selectedIndex = i;
                    else if (destName === "Varanasi" && options[i].value.includes("VNS")) toSelect.selectedIndex = i;
                    else if (destName === "Kochi" && options[i].value.includes("COK")) toSelect.selectedIndex = i;
                }
                document.getElementById('bookingPanel').scrollIntoView({ behavior: 'smooth' });
                const msgDiv = document.getElementById('flightResultMsg');
                msgDiv.style.display = 'block';
                msgDiv.innerHTML = `✨ Destination set to ${destName}. Select dates to book flights.`;
                msgDiv.style.background = "#fff0e6";
                setTimeout(() => {
                    msgDiv.style.opacity = '0';
                    setTimeout(() => { msgDiv.style.display = 'none'; msgDiv.style.opacity = '1'; }, 1000);
                }, 2000);
            });
        });
    }

    // handle flight search form
    const flightForm = document.getElementById('flightSearchForm');
    const resultMsg = document.getElementById('flightResultMsg');

    function generateFlightResults(from, to, depart, returnDate, paxCount) {
        const airlines = ['IndiGo', 'Air India', 'SpiceJet', 'Akasa Air', 'Vistara'];
        const randomAirline = airlines[Math.floor(Math.random() * airlines.length)];
        const priceBase = Math.floor(Math.random() * (4500 - 2100 + 1) + 2100);
        const totalPrice = priceBase * (paxCount === '1 Adult' ? 1 : paxCount === '2 Adults' ? 2 : paxCount === '3 Adults' ? 3 : 4);
        const flightNo = Math.floor(Math.random() * 9000 + 1000);
        const departureTime = `${Math.floor(Math.random() * 12 + 5)}:${Math.floor(Math.random() * 60)} ${Math.random() > 0.5 ? 'AM' : 'PM'}`;
        if (returnDate) {
            return `✈️ ${randomAirline} ${flightNo} · ${from} → ${to} · Dep ${departDate} · Return ${returnDate} · ${paxCount} · Total ₹${totalPrice} (incl. taxes). Great deal! Book now.`;
        } else {
            return `✈️ ${randomAirline} ${flightNo} · One-way from ${from} to ${to} on ${departDate} · ${paxCount} · ₹${priceBase} per person | Total ₹${totalPrice}. Secure your seat!`;
        }
    }

    flightForm.addEventListener('submit', function(e) {
        e.preventDefault();

        const fromCity = document.getElementById('fromCity').value;
        const toCity = document.getElementById('toCity').value;
        const departDate = document.getElementById('departDate').value;
        const returnDate = document.getElementById('returnDate').value;
        const passengers = document.getElementById('passengers').value;

        if (!fromCity || !toCity) {
            showMsg("Please select departure and arrival cities.", true);
            return;
        }
        if (fromCity === toCity) {
            showMsg("Origin and destination cannot be same. Choose different cities.", true);
            return;
        }
        if (!departDate) {
            showMsg("Select departure date.", true);
            return;
        }
        const today = new Date().toISOString().split('T')[0];
        if (departDate < today) {
            showMsg("Departure date cannot be in the past.", true);
            return;
        }
        if (returnDate && returnDate < departDate) {
            showMsg("Return date must be after departure date.", true);
            return;
        }

        const flightInfo = generateFlightResults(fromCity, toCity, departDate, returnDate, passengers);
        showMsg(flightInfo, false);

        // Optionally reset form partially? we keep but show confirmation
    });

    function showMsg(msg, isError) {
        resultMsg.style.display = 'block';
        resultMsg.innerHTML = `<i class="fas fa-plane"></i> ${msg}`;
        resultMsg.style.backgroundColor = isError ? '#ffe8e6' : '#e3f2fd';
        resultMsg.style.color = isError ? '#b33c2e' : '#145c8f';
        setTimeout(() => {
            resultMsg.style.opacity = '0';
            setTimeout(() => {
                resultMsg.style.display = 'none';
                resultMsg.style.opacity = '1';
            }, 1000);
        }, 5000);
    }

    // set min dates
    const todayStr = new Date().toISOString().split('T')[0];
    const departInput = document.getElementById('departDate');
    const returnInput = document.getElementById('returnDate');
    if (departInput) departInput.min = todayStr;
    if (returnInput) returnInput.min = todayStr;

    // scroll to booking on click
    document.getElementById('scrollToBooking')?.addEventListener('click', () => {
        document.getElementById('bookingPanel').scrollIntoView({ behavior: 'smooth' });
    });

    // also attach quick select from destinations
    renderDestinations();
</script>
