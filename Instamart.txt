    function updateCartUI() {

    function attachCartItemEvents() {
        document.querySelectorAll('.decrCart').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.dataset.id);
                const idx = cart.findIndex(i => i.id === id);
                if (idx !== -1) {
                    if (cart[idx].quantity > 1) cart[idx].quantity--;
                    else cart.splice(idx, 1);
                    updateCartUI();
                    saveCartToLocal();
                    renderProducts(); // re-render to keep button states fresh
                }
            });
        });
        document.querySelectorAll('.incrCart').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.dataset.id);
                const idx = cart.findIndex(i => i.id === id);
                if (idx !== -1) cart[idx].quantity++;
                updateCartUI();
                saveCartToLocal();
                renderProducts();
            });
        });
        document.querySelectorAll('.removeCartItem').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.dataset.id);
                cart = cart.filter(i => i.id !== id);
                updateCartUI();
                saveCartToLocal();
                renderProducts();
            });
        });
    }

    function addToCart(product) {
        const existing = cart.find(i => i.id === product.id);
        if (existing) existing.quantity++;
        else cart.push({ id: product.id, name: product.name, price: product.price, quantity: 1 });
        updateCartUI();
        saveCartToLocal();
        renderProducts(); // reflect button state (optional: change add button to "added")
    }

    function saveCartToLocal() { localStorage.setItem('instamart_cart', JSON.stringify(cart)); }
    function loadCart() {

    }

        document.querySelectorAll('.add-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.dataset.id);
                const prod = products.find(p => p.id === id);
                if (prod) addToCart(prod);
            });
        });
        document.querySelectorAll('.quick-incr').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.dataset.id);
                const prod = products.find(p => p.id === id);
                if (prod) addToCart(prod);
            });
        });
        document.querySelectorAll('.quick-decr').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.dataset.id);
                const idx = cart.findIndex(i => i.id === id);
                if (idx !== -1) {
                    if (cart[idx].quantity > 1) cart[idx].quantity--;
                    else cart.splice(idx, 1);
                    updateCartUI();
                    saveCartToLocal();
                    renderProducts();
                }
            });
        });
        document.querySelectorAll('.quick-remove').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const id = parseInt(btn.dataset.id);
                cart = cart.filter(i => i.id !== id);
                updateCartUI();
                saveCartToLocal();
                renderProducts();
            });
        });
    }

    // category filter handling
    function setActiveCategory(cat) {
        activeCategory = cat;
        categoryTabs.forEach(tab => {
            if (tab.dataset.cat === cat) tab.classList.add('active');
            else tab.classList.remove('active');
        });
        renderProducts();
    }
    categoryTabs.forEach(tab => {
        tab.addEventListener('click', () => {
            setActiveCategory(tab.dataset.cat);
        });
    });

    sortSelect.addEventListener('change', () => renderProducts());
    resetBtn.addEventListener('click', () => {
        searchInput.value = "";
        searchTerm = "";
        sortSelect.value = "default";
        setActiveCategory("all");
        renderProducts();
    });

        if (cart.length === 0) alert("Your cart is empty! Add some products.");
        else {
            alert(`✅ Order placed! Total ₹${cart.reduce((s,i)=>s+(i.price*i.quantity),0)}. Delivery in 15 min 🚀`);
            cart = [];
            updateCartUI();
            saveCartToLocal();
            renderProducts();
            closeCart();
        }
    });
