<!-- simple interactive javascript for application simulation -->
<script>
    (function() {
        const form = document.getElementById('bankApplicationForm');
        const statusDiv = document.getElementById('formStatus');

        function showMessage(msg, isError = false) {
            statusDiv.style.display = 'block';
            statusDiv.innerHTML = msg;
            statusDiv.style.backgroundColor = isError ? '#ffe6e5' : '#e3f5e9';
            statusDiv.style.color = isError ? '#b33c2e' : '#1f7640';
            statusDiv.style.borderLeft = `4px solid ${isError ? '#d9534f' : '#2b8c4a'}`;
            setTimeout(() => {
                if(statusDiv.style.display === 'block') {
                    // fade out slowly but keep visible for few secs
                    setTimeout(() => {
                        statusDiv.style.opacity = '0';
                        setTimeout(() => {
                            if(statusDiv.style.display !== 'none') statusDiv.style.display = 'none';
                            statusDiv.style.opacity = '1';
                        }, 400);
                    }, 2800);
                }
            }, 3000);
        }

        form.addEventListener('submit', function(e) {
            e.preventDefault();

            const fullName = document.getElementById('fullName').value.trim();
            const email = document.getElementById('email').value.trim();
            const accountType = document.getElementById('accountType').value;
            const deposit = parseFloat(document.getElementById('deposit').value);

            // simple validation
            if (!fullName) {
                showMessage("❌ Please provide your full name.", true);
                return;
            }
            if (!email || !email.includes('@') || !email.includes('.')) {
                showMessage("⚠️ Enter a valid email address.", true);
                return;
            }
            if (isNaN(deposit) || deposit < 50) {
                showMessage("💰 Minimum initial deposit is $50. Please adjust.", true);
                return;
            }

            // simulate success response
            const refNumber = 'PT' + Math.floor(Math.random() * 1000000);
            const message = `🎉 Thanks ${fullName}! Your ${accountType} application is received. Reference: ${refNumber}. We'll contact you within 1 business day at ${email}.`;
            showMessage(message, false);
            // reset form fields optional but nice to keep for demo
            // optionally reset but we keep data for demonstration
            document.getElementById('fullName').value = '';
            document.getElementById('email').value = '';
            document.getElementById('deposit').value = '500';
            document.getElementById('accountType').selectedIndex = 0;
            // Extra: optional dynamic reset but user can apply again
        });
    })();
</script>
