    let currentCountryFilter = "all";
    let currentSearch = "";

    function renderBrands() {
        let filtered = carBrands.filter(brand => {
            const matchCountry = currentCountryFilter === "all" || brand.country === currentCountryFilter;
            const matchSearch = brand.name.toLowerCase().includes(currentSearch.toLowerCase());
            return matchCountry && matchSearch;
        });
        const grid = document.getElementById("brandsGrid");
        if (filtered.length === 0) {
            grid.innerHTML = `<div style="grid-column:1/-1; text-align:center; padding:50px;"><i class="fas fa-car-crash"></i> No brands found. Try different filter.</div>`;
            return;
        }
        grid.innerHTML = filtered.map(brand => `
            <div class="brand-card" data-id="${brand.id}">
                <div class="car-image" style="background-image: url('${brand.carImage}');">
                    <div class="brand-logo-overlay">
                        <img src="${brand.logo}" alt="${brand.name} logo" onerror="this.src='https://via.placeholder.com/48?text=Logo'">
                    </div>
                </div>
                <div class="brand-info">
                    <div class="brand-name">${brand.name}</div>
                    <div class="brand-country"><i class="fas fa-flag-checkered"></i> ${brand.country}</div>
                    <div class="brand-cars">🏎️ ${brand.models.split(',')[0]} ...</div>
                </div>
            </div>
        `).join('');
        // attach click events for modal
        document.querySelectorAll('.brand-card').forEach(card => {
            card.addEventListener('click', (e) => {
                const id = parseInt(card.dataset.id);
                const brand = carBrands.find(b => b.id === id);
                if(brand) openBrandModal(brand);
            });
        });
    }

    // Modal logic with car image
    function openBrandModal(brand) {
        document.getElementById("modalBrandName").innerText = brand.name;
        document.getElementById("modalCountry").innerHTML = `<i class="fas fa-globe"></i> ${brand.country}`;
        document.getElementById("modalFounded").innerHTML = `Founded: ${brand.founded}`;
        document.getElementById("modalDesc").innerHTML = brand.description;
        
        // Set modal car image
        const modalCarDiv = document.getElementById("modalCarImage");
        modalCarDiv.style.backgroundImage = `url('${brand.carImage}')`;
        modalCarDiv.style.backgroundSize = "cover";
        modalCarDiv.style.backgroundPosition = "center";
        
        const modalLogoDiv = document.getElementById("modalLogo");
        modalLogoDiv.innerHTML = `<img src="${brand.logo}" alt="${brand.name}" style="max-width:80px;" onerror="this.src='https://via.placeholder.com/80?text=Logo'">`;
        
        document.getElementById("modalExploreBtn").onclick = () => {
            alert(`🚗 Explore ${brand.name} models: ${brand.models}. Visit official website for more details.`);
        };
        document.getElementById("brandModal").classList.add("active");
    }
    
    function closeModal() { 
        document.getElementById("brandModal").classList.remove("active"); 
    }

    // filter events
    document.getElementById("searchBrand").addEventListener("input", (e) => {
        currentSearch = e.target.value;
        renderBrands();
    });
    document.getElementById("countryFilter").addEventListener("change", (e) => {
        currentCountryFilter = e.target.value;
        renderBrands();
    });
    document.getElementById("resetFiltersBtn").addEventListener("click", () => {
        document.getElementById("searchBrand").value = "";
        document.getElementById("countryFilter").value = "all";
        currentSearch = "";
        currentCountryFilter = "all";
