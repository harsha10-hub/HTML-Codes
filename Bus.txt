    // Helper: generate seat labels (A1..J8 dynamic per bus)
    function generateSeatLabels(bus) {
        const total = bus.seatsTotal;
        let letters = ["A","B","C","D","E","F","G","H","I","J"];
        let seats = [];
        let count = 1;
        for(let i=0; i<letters.length && seats.length < total; i++) {
            for(let s=1; s<=6; s++) {
                if(seats.length >= total) break;
                seats.push(`${letters[i]}${s}`);
            }
        }
        return seats.slice(0, total);
    }

    // Open seat selection modal
    function openSeatModal(bus) {
        currentSelectedBus = bus;
        selectedSeatsArray = [];
        const modalTitle = document.getElementById("modalBusTitle");
        modalTitle.innerText = `${bus.name} (${bus.from} → ${bus.to})`;
        renderSeatMap(bus);
        updateSeatModalSummary();
        document.getElementById("seatModal").classList.add("active");
    }

    function renderSeatMap(bus) {
        const allSeats = generateSeatLabels(bus);
        const booked = bus.bookedSeats || [];
        const seatContainer = document.getElementById("seatMapGrid");
        seatContainer.innerHTML = "";
        allSeats.forEach(seat => {
            const isBooked = booked.includes(seat);
            const isSelected = selectedSeatsArray.includes(seat);
            const seatDiv = document.createElement("div");
            seatDiv.classList.add("seat");
            if (isBooked) seatDiv.classList.add("booked");
            if (isSelected) seatDiv.classList.add("selected");
            seatDiv.innerText = seat;
            if (!isBooked) {
                seatDiv.classList.add("available");
                seatDiv.addEventListener("click", () => {
                    if (selectedSeatsArray.includes(seat)) {
                        selectedSeatsArray = selectedSeatsArray.filter(s => s !== seat);
                    } else {
                        selectedSeatsArray.push(seat);
                    }
                    renderSeatMap(bus);
                    updateSeatModalSummary();
                });
            }
            seatContainer.appendChild(seatDiv);
        });
    }

    function updateSeatModalSummary() {
        const count = selectedSeatsArray.length;
        const total = count * currentSelectedBus.price;
        document.getElementById("selectedSeatsLabel").innerText = count ? selectedSeatsArray.join(", ") : "None";
        document.getElementById("totalFareSpan").innerText = total;
    }

    function confirmBusBooking() {
        if (!currentSelectedBus) return;
        if (selectedSeatsArray.length === 0) {
            alert("Please select at least one seat.");
            return;
        }
        // check double booking conflict
        const conflict = selectedSeatsArray.some(seat => currentSelectedBus.bookedSeats.includes(seat));
        if (conflict) {
            alert("One or more selected seats just got booked. Please reselect.");
            openSeatModal(currentSelectedBus);
            return;
        }
        // create booking record
        const bookingRecord = {
            id: Date.now(),
            busId: currentSelectedBus.id,
            busName: currentSelectedBus.name,
            from: currentSelectedBus.from,
            to: currentSelectedBus.to,
            date: document.getElementById("travelDate").value || "2026-05-25",
            seats: [...selectedSeatsArray],
            totalPrice: selectedSeatsArray.length * currentSelectedBus.price,
            bookedOn: new Date().toLocaleString()
        };
        userBookings.push(bookingRecord);
        localStorage.setItem("yatri_bookings", JSON.stringify(userBookings));

        // update current bus booked seats (persist)
        currentSelectedBus.bookedSeats.push(...selectedSeatsArray);
        // also update in original busesData reference
        const index = busesData.findIndex(b => b.id === currentSelectedBus.id);
        if (index !== -1) busesData[index].bookedSeats = currentSelectedBus.bookedSeats;
        // store updated bus seat data to localStorage for future sessions
        localStorage.setItem("yatri_buses_state", JSON.stringify(busesData.map(b => ({ id: b.id, bookedSeats: b.bookedSeats }))));

        alert(`✅ Booking confirmed!\nBus: ${currentSelectedBus.name}\nSeats: ${selectedSeatsArray.join(", ")}\nTotal: ₹${selectedSeatsArray.length * currentSelectedBus.price}\nJourney: ${currentSelectedBus.from} → ${currentSelectedBus.to} on ${bookingRecord.date}`);
        
        closeModalAndRefresh();
    }

    function closeModalAndRefresh() {
        document.getElementById("seatModal").classList.remove("active");
        currentSelectedBus = null;
        selectedSeatsArray = [];
        filterAndRenderBuses();  // refresh bus list (seat availability updated)
    }

    }

                if (bus) bus.bookedSeats = updated.bookedSeats;
            });
        }
        const storedBookings = localStorage.getItem("yatri_bookings");
        if (storedBookings) userBookings = JSON.parse(storedBookings);
    }
